model1_log = smf.mixedlm("log_NO2_emission ~ NDVI + Park_size + Distance_to_Park + Temperature + Wind_speed+ NDVI_Temp_Interaction + NDVI_Wind_speed_Interaction",

result1_log = model1_log.fit()
